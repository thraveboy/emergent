def clear_screen(log = "", full_wipe = false)
  if $dump_logs
    if (log == "map")
      if full_wipe
        $team_log.print("\e[2J")
      else
        # go to top left corner instead of clear
        $map_log.print("\e[2H")
      end
    elsif (log == "team")
      $team_log.print("\e[2H")
    else
      print "\e[2J"
    end
  end
end

def output_team_stats(beings, world, output_to_file=false)
  team_results = TeamStats_BeingOperator.new.execute(beings, world, output_to_file)
  clear_screen("team")
  team_results.each do |key, value|
    putsl("Team #{key}: #{value}", "team")
    putsl("", "team")
  end
end

def print_things_list(name, things, displays = [])
  i = 0
  putsl "#{name}s in existence..."
  putsl "-"
  putsl "_"
  for thing_instance in things
    i += 1
    printl "#{name} #{i} : "
    thing_instance.print_this_baby_out([], displays)
  end
  putsl "_"
  putsl "-"
end

def printl (text, log_file = "battle")
  if $dump_logs
    print text
    if log_file == "battle"
      $battle_log.write text
    elsif log_file == "team"
     $team_log.write text
    else
      $map_log.write text
    end
  end
end
