def initialize(filename = nil)
  if filename == nil
    @initialized = true
    return
  end
  if File.exists?(filename)
    File.read(filename).each_line do |line|
      if !line.nil?
        self.addCharacteristic(line.rstrip)
      end
    end
    @initialized = true
    @filename = filename
  else
    putsl red("..._")
    putsl red("   file #{filename} does not exist__")
    putsl red("                                    <<<.")
  end
end

def addCharacteristic (characteristic_line)
  split_line =  characteristic_line.split(' ',2)
  key = split_line[0]
  value = split_line[1]
  self.class.send(:attr_accessor, key)
  instance_variable_set("@#{key}", value)
rescue
  puts red("Can not add characteristic from line #{characteristic_line}")
end

def print_this_baby_out(omitted_variables = [], displays=[])
  putsl @name
  cray_cray = ""
  stuck_middle = ":"
  coo_coo   = ""
  instance_variables.each do |iv|
    if !(omitted_variables.include? (iv[1..-1]))
      iv_value = instance_variable_get(iv)
      iv_jv_name = "#{iv}"
      iv_jv_name[0] = ''
      putsl "#{cray_cray}#{iv_jv_name}#{stuck_middle}#{iv_value}#{coo_coo}"
    end
  end
end
