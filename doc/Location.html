def can_add?(what_to_add)
  allowed_types = ["being"]
  obj_type = nil
  if !(what_to_add.nil?)
    if allowed_types.include?(what_to_add.class.name.downcase)
      obj_type = what_to_add.class.name.downcase
    end
  end
  if !(obj_type.nil?)
    current_objs = instance_variable_get("@#{obj_type}s")
    if (current_objs == nil) || (current_objs == [])
      return true
    end
  end
  return false
end
